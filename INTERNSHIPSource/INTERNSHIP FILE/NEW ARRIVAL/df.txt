            const angle = (i / data.length) * (2 * Math.PI);

            const node = document.createElement('div');
            node.className = 'node';
            node.style.left = x + 'px';
            node.style.top = y + 'px';

            node.innerHTML = `
                <div class="circle">${i + 1}</div>
                <div class="label">${text}</div>
            `;

            container.appendChild(node);
